/// <reference path='../extern/express.d.ts'/>

/**
 * @namespace TizenApplicationStubService
 */

import abstractService = module("abstractService");
import packageRepositoryService = module("packageRepositoryService");
import express = module("express");
import eventSourceDispatcher = module("../eventSourceDispatcher");

/** 
* `/tizenApplication`
*
* Declares Tizen Application API as stub service for IVI environment which doesn't include Tizen Application APIs at the moment.
* @class TizenApplicationStubService
* @extends AbstractService.AbstractService
* @constructor
*/
/** 
* Event fired after new application was installed.
* @event installed
* @for TizenApplicationStubService
* @param eventArgs {InstalledPackageData} Event arguments describing installed application.
*/
export class TizenApplicationStubService extends abstractService.AbstractService {
    /** 
    * @method constructor
    * @param packageRepository {PackageRepositoryService} Instance of `PackageRepositoryService`
    * @param eventSourceDispatcher {EventSourceDispatcher} Instance of `EventSourceDispatcher`
    * @param launchWithPersistentConsole {Boolean} Indicates if applications should be launched with HTML console messages stored in application log
    */
    constructor (private packageRepository: packageRepositoryService.PackageRepositoryService,
                 private eventSourceDispatcher: eventSourceDispatcher.EventSourceDispatcher, private launchWithPersistentConsole: bool) {
        super({
            name: "TizenApplicationStubService",
            description: "Tizen Application Stub",
            endpointName: "tizenApplication",
            apiClassName: "application",
            apiView: "fakeTizenApplicationApi",
            eventSourceDispatcher: eventSourceDispatcher
        });
    }

    attachApplication(aApplication: express.ExpressServer) {
        var self: TizenApplicationStubService = this;
        this.actions.splice(this.actions.length, 0, {
            type: "navigate",
            title: "Launch application",
            apiMethod: "launch",
            endpoint: "/launch",
            callback: this.launch,
            parameters: [{
                name: "id",
                type: "text",
                value: ""
            }]
        }, {
            type: "navigate",
            title: "Hide application",
            apiMethod: "hide",
            endpoint: "/hide",
            callback: this.hide,
            parameters: []
        }, {
            type: "navigate",
            title: "Exit application",
            apiMethod: "exit",
            endpoint: "/exit",
            callback: this.exit,
            // Same effect for us :)
            parameters: []
        }, {
            type: "get",
            title: "Get installed application info",
            endpoint: "/getAppsInfo",
            apiMethod: "getAppsInfo",
            callback: this.get
        }, {
            type: "event",
            title: "AddApplicationInstalledListener",
            apiMethod: "__addApplicationInstalledListener",
            message: "listener"
        }
        );

        this.packageRepository.on("changed", function (aArgs) {
            try {
                if (aArgs.type === "installedApplication") {
                    self.logger.info("Detected application change: %j", aArgs);
                    self.packageRepository.getInstalledApps().forEach(function (aItem, aIndex) {
                        if (aItem.__applicationDirectory == aArgs.application) {
                            self.logger.info("New application installed: %j", aItem);
                            self.emit("installed", aItem);
                        }
                    });
                }
            } catch (ex) {
                self.logger.error(ex);
            }
        });

        this.on("installed", function (aAppInfo) {
            self.sendStatus("listener", aAppInfo);
        });

        super.attachApplication(aApplication);
    }

    /**
    * `[GET] /getAppsInfo` Returns information about installed applications
    * @method get
    * @return {InstalledPackageData[]} Array of installed applications
    */
    get(aRequest: express.ExpressServerRequest, aResponse: express.ExpressServerResponse) {
        this.logger.info("Updating list of applications");
        var applications = this.packageRepository.getInstalledApps();

        applications.forEach(function (aApp) {
            aApp.iconPath = "../" + aApp.__applicationDirectory + "/" + aApp.iconPath;
        });

        aResponse.json(applications);
    }

    /**
    * `[NAVIGATE] /launch` Launches specified application. Replacement of tizen.application.launch() call.
    * @method launch
    * @param id {String} Application ID, e.g. http://com.intel.tizen/store
    */
    launch(aRequest: express.ExpressServerRequest, aResponse: express.ExpressServerResponse) {
        var applicationId: string = aRequest.query.id;
        this.logger.info("Launching app %s", applicationId);
        var apps = this.packageRepository.getInstalledApps();
        var redirectUrl;

        for (var i in apps) {
            if (apps[i].id === applicationId) {
                redirectUrl = "../../application/" + apps[i].__applicationDirectory + "/index.html"
                redirectUrl += this.launchWithPersistentConsole ? "?persistentConsole" : "";
                this.logger.info("redirecting to " + redirectUrl);
                break;
            }
        }

        this.logger.info("Redirecting to %s", apps[i].id);
        aResponse.redirect(redirectUrl);
    };

    /**
    * `[NAVIGATE] /hide` Hides current application. Replacement of tizen.application.hide() call.
    * @method hide
    */
    hide(aRequest: express.ExpressServerRequest, aResponse: express.ExpressServerResponse) {
        this.logger.info("Hiding app");

        var apps = this.packageRepository.getInstalledApps();
        var redirectUrl = "../../application/com.intel.tizen_homescreen/index.html";

        for (var i in apps) {
            if (apps[i].__applicationDirectory === "homescreen") {
                redirectUrl = "../../application/homescreen/index.html";
                break;
            }
        }

        this.logger.info("redirecting to " + redirectUrl);
        aResponse.redirect(redirectUrl);
    }

    /**
    * `[NAVIGATE] /exit`
    * Exists current application. Replacement of tizen.application.exit() call.
    * @method exit
    */
    exit(aRequest: express.ExpressServerRequest, aResponse: express.ExpressServerResponse) {
        this.logger.info("Exiting app");
        var apps = this.packageRepository.getInstalledApps();
        var redirectUrl = "../../application/com.intel.tizen_homescreen/index.html";

        for (var i in apps) {
            if (apps[i].__applicationDirectory === "homescreen") {
                redirectUrl = "../../application/homescreen/index.html";
                break;
            }
        }

        this.logger.info("redirecting to " + redirectUrl);
        aResponse.redirect(redirectUrl);
    }
}
