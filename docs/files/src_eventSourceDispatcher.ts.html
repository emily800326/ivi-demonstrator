/// <reference path='./extern/node.d.ts'/>
/// <reference path='./extern/express.d.ts'/>
/// <reference path='./extern/fakeEmitter.d.ts'/>

import util = module("util");
import express = module("express");

var log4js = require('log4js'),
    _EventEmitter = require("events").EventEmitter;

/**
 * @namespace EventSourceDispatcher
 */

/** 
* Wrapper for event data passed to HTML5 applications.
* @class EventUpdateData
*/
interface EventUpdateData {
    /** 
    * Optional message identificator
    * @property id
    * @optional
    * @type {String}
    */
    id?: string;
    /** 
    * Message source descriptor
    * @property source
    * @type {String}
    */
    source: string;
    /** 
    * Message payload
    * @property message
    * @type {Object}
    */
    message: any;
}

/** 
* Declares single point of dispatching HTML5 `EventSource` messages using `text/event-stream` content type to avoid WebKit limitations. 
* This class works as server-side complementary component to class `ApiEndpointBase`.
* WebKit engine currently supports up to 5 parallel `EventSource` sources. This can be avoided by declaring one common source and dispatching 
* messages based on identificator. To send new message use emit method e.g.:
* 
*      eventSourceDispatcher.emit("send", { source: "mysource", message: "Test message" });
*
* @class EventSourceDispatcher
* @extends EventEmitter
* @constructor
*/
export class EventSourceDispatcher extends _EventEmitter {
    public logger: any;
    private firstTime: bool = true;

    /** 
    * @method constructor 
    */
    constructor () {
        super();
        this.logger = log4js.getLogger("EventSourceDispatcher");
    }

    /** 
    * Emit new message to HTML5 application.
    * @method emit
    * @param name {String} Name of the action. Should be "send".
    * @param data {EventSourceDispatcher.EventUpdateData} Event data.
    */

    /** 
    * Attach this class to Express HTTP server application.
    * @method attachApplication
    * @param aApplication {ExpressApplication} Instance of Express application
    */
    attachApplication(aApplication: any) {
        aApplication.get('/ivi-events', this.getEvents.bind(this));
    }

    /** 
    * `[EVENT]` Declares API endpoint for receiving events via `EventSource` mechanism.
    * @method getEvents
    */
    getEvents(aRequest: express.ExpressServerRequest, aResponse: any) {
        var self = this;
        var referer = aRequest.headers["referer"];

        this.logger.info("EventSource stream requested by %s", referer);

        aResponse.header('Content-Type', 'text/event-stream');
        aResponse.header('Cache-Control', 'no-cache');
        aResponse.header('Connection', 'keep-alive');
        aResponse.connection.setTimeout(0);

        var responseFunction = function (aEventData: EventUpdateData) {
            self.logger.debug("Sending message from %s to %s: %j", aEventData.source, referer, aEventData.message);
            var payload = util.format("id: %d\n", new Date().toLocaleTimeString());
            payload += util.format("data: %j\n\n", aEventData);

            aResponse.write(payload);
        };

        this.on("send", responseFunction);

        if (this.firstTime) {
            responseFunction({ source: "global", message: "initialized" });
            this.firstTime = false;
        }

        aRequest.connection.addListener('close', function () {
            self.logger.info("EventSource stream closed by client %s, closing response ...", referer);
            aResponse.end("\n");
            self.removeListener("send", responseFunction);
        });
    }
}
