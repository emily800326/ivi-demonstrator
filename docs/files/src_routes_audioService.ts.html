import abstractService = module("abstractService");
import express = module("express");

/**
 * @namespace AudioService
 */

import eventSourceDispatcher = module("../eventSourceDispatcher");

/**
* Wraps status of audio service
* @class StatusData
*/
interface StatusData {
    /** 
    * Indicates if song playback should repeat.
    * @property repeat
    * @type {Boolean}
    */
    repeat: bool;
    /** 
    * Indicates if song playback should shuffle.
    * @property shuffle
    * @type {Boolean}
    */
    shuffle: bool;
    /** 
    * Indicates if song is playing or it's stopped.
    * @property play
    * @type {Boolean}
    */
    play: bool;
    indexInPlayList: number;
    position: number;
    volume: number;
}

/** 
* Declares Audio Service API which provides methods to one audio resource from multiple frontend application demonstrating this concept .
* Notification are sent to listeners by HTML5 `EventSource` mechanism available in class `ThemeEngine.addStatusListener` e.g.:
*
*     var audioService = new AudioService ();   
*     audioService.addStatusListener(function (data) { 
*         // Process changes
*     });
*
* @class AudioService
* @extends AbstractService.AbstractService
* @constructor
*/
/** 
* Event fired after one or more audio service status changed.
* @event changed
* @param aData {AudioService.StatusData} New audio status
* @for AudioService
*/
export class AudioService extends abstractService.AbstractService {
    private status: StatusData;

    /** 
    * @method constructor
    * @param eventSourceDispatcher {EventSourceDispatcher.EventSourceDispatcher} Instance of `EventSourceDispatcher`
    */
    constructor (eventSourceDispatcher: eventSourceDispatcher.EventSourceDispatcher) {
        super({
            name: "AudioService",
            description: "Audio Service",
            apiClassName: "AudioService",
            indexView: "audioService",
            eventSourceDispatcher: eventSourceDispatcher
        });

        this.status = {
            repeat: false,
            shuffle: false,
            play: false,
            indexInPlayList: 1,
            position: 0,
            volume: 50
        };
    }

    attachApplication(aApplication: any) {
        var self = this;
        this.actions.splice(this.actions.length, 0, {
            type: "event",
            title: "Indicator live status",
            apiMethod: "addStatusListener",
            message: "status",
        }, {
            type: "get",
            title: "Get status",
            apiMethod: "getStatus",
            endpoint: "/status",
            callback: this.getStatus
        },
        {
            type: "post",
            title: "Set status",
            apiMethod: "setStatus",
            endpoint: "/status",
            callback: this.postStatus,
            parameters: [{
                name: "indicator",
                type: "text",
                value: "repeat"
            }, {
                name: "value",
                type: "text",
                value: "false"
            }]
        },
        {
            type: "post",
            title: "Set status all",
            apiMethod: "setStatusAll",
            endpoint: "/statusAll",
            callback: this.postStatusAll,
            parameters: [{
                name: "value",
                type: "text",
                value: JSON.stringify(this.status)
            }]
        }
            );
        super.attachApplication(aApplication);
        this.on("changed", function (aStatus) {
            self.sendStatus("status", aStatus);
        });
    }

    /**
    * `[GET] /status` Returns current status
    * @method getStatus
    * @return {StatusData}
    */
    getStatus(aRequest: express.ExpressServerRequest, aResponse: express.ExpressServerResponse) {
        aResponse.json(this.status);
    }

    /**
    * `[POST] /status` Set single audio status indicator
    * @method postStatus
    * @param indicator {String} Indicator name
    * @param status {String} Indicator value
    */
    postStatus(aRequest: express.ExpressServerRequest, aResponse: express.ExpressServerResponse) {
        this.logger.info("postStatus called");
        var self = this;

        if (this.status[aRequest.body.value] != "") {
            this.status[aRequest.body.indicator] = aRequest.body.value;
        }
        this.emit('changed', this.status);

        this.getStatus(aRequest, aResponse);
    }

    /**
    * `[POST] /statusAll` Set all audio status indicators
    * @method postStatusAll
    * @param value {AudioService.StatusData} Indicator name
    */
    postStatusAll(aRequest: express.ExpressServerRequest, aResponse: express.ExpressServerResponse) {
        this.logger.info("new status: %s", aRequest.body.value);

        var self = this;

        if ((typeof aRequest.body.value).toLowerCase() === "string") {
            try {
                this.status = JSON.parse(aRequest.body.value)
            } catch (error) {
                this.logger.info("Unable to parse status: %s", error.message);
            }
        } else {
            this.status = aRequest.body.value;
        }

        this.logger.info("this status: %s", JSON.stringify(this.status));
        this.emit('changed', this.status);
        this.getStatus(aRequest, aResponse);
    }
