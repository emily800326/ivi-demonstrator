interface IEventSource {
    addEventListener: (source: string, callback, flag: bool) => void;
    onmessage: (callback) => void;
}

/**
 * @namespace ApiEndpointBase
 */

interface Window {
    EventSource: IEventSource;
}

declare var window: Window;
declare var EventSource: any;

/** 
* Declares single point of dispatching HTML5 EventSource messages using `text/event-stream` content type to avoid WebKit limitations. 
* This class works as client-side complementary component to class `EventSourceDispatcher.EventSourceDispatcher`.
* WebKit engine currently supports up to 5 parallel `EventSource` sources. This can be avoided by declaring one common source and dispatching 
* messages based on identificator. To receive push messages from server use `addIviEventSourceListener` method:
* 
*     addIviEventSourceListener(8088, "mysource", function(aMessage) { 
*       // Process message
*     });
* 
* @class ApiEndpointBase
*/
module ApiEndpointBase {
    private source: IEventSource;
    private listeners = [];

    function eventSourceCallback(e) {
        try {
            var dataObject = (typeof e.data == 'string') ? JSON.parse(e.data) : e.data;
            console.info("Message arrived:", dataObject)

            if (dataObject.source == "global") {
                console.info("Global message arrived:" + JSON.stringify(dataObject.message));
            } else {
                var found = false;
                listeners.forEach(function (listener) {
                    if (listener.source == dataObject.source) {
                        listener.callback(dataObject.message);
                        found = true;
                    }
                });

                if (!found) {
                    console.warn("No EventSource listener found for source: " + dataObject.source);
                }
            }
        } catch (ex) {
            console.error(ex);
        }
    }

    /**
    * Adds new listener to message source.
    * @method addIviEventSourceListener
    * @async
    * @param aPort {Integer} Server port number
    * @param aMessageSource {String} Message source as defined by EventUpdateData.source
    * @param aCallback {Callback} Method processing server side push messages
    */
    export function addIviEventSourceListener(aPort: number, aMessageSource: string, aCallback) {
        if (!source) {
            var basePath = window.location.protocol + "//" + window.location.host + "/ivi-events?salt=" + new Date().getTime();
            console.info("First call to service, creating EventSource stream for: " + basePath);

            if (!!window.EventSource) {
                source = new EventSource(basePath);
                source.onmessage = eventSourceCallback;
            } else {
                console.error("EventSource is not supported.");
            }
        }

        var firstInstance = true;

        listeners.forEach(function (listener) {
            if (listener.source == aMessageSource) {
                firstInstance = false;
                return true;
            }
        });

        listeners.push({ source: aMessageSource, callback: aCallback });
        return firstInstance;
    }
}
